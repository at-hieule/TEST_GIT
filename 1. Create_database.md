*Create table user*

```
CREATE TABLE `blog`.`user` ( `id` INT(11) NOT NULL, `full_name` VARCHAR(255) NOT NULL, `email` VARCHAR(255) NULL,`rank` TINYINT(4) NOT NULL, `is_active` TINYINT(1) NOT NULL, `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`));

```


*Create tab category*

```
CREATE TABLE `blog`.`category` ( `id` INT(11) NOT NULL, `title` VARCHAR(255) NOT NULL, `description` VARCHAR(255) NULL, PRIMARY KEY (`id`));

```

*Create tab blog*

```
CREATE TABLE `blog`.`blog` ( `id` INT(11) NOT NULL, `category_id` INT(11) NULL, `user_id` INT(11) NULL, `title` VARCHAR(255) NULL, `view` INT(11) NULL, `is_active` TINYINT(1) NULL, `content` TEXT  NULL, `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `updated_at` TIMESTAMP NOT NULL ON UPDATE CURRENT_TIMESTAMP , PRIMARY KEY (`id`));

```

*Create table news*

```
CREATE TABLE `blog`.`news` ( `id` INT(11) NOT NULL, `category_id` INT(11) NOT NULL,`title` VARCHAR(255) NOT NULL, `view` INT(11) NULL,`is_active` TINYINT(1) NOT NULL, `content` TEXT NOT NULL,`created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `updated_at` TIMESTAMP NOT NULL ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`id`));

```


*Create table follow*

```
CREATE TABLE `blog`.`follow` ( `id` INT(11) NOT NULL, `from_user_id` INT(11) NOT NULL, `to_user_id` INT(11) NOT NULL, `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (`id`));

```

*Create table comment*

CREATE TABLE `blog`.`comment` ( `id` INT(11) NOT NULL, `taget_table`  VARCHAR(20) NULL,`target_id` INT(11) NOT NULL, `user_id` INT(11) NOT NULL,`comment` TEXT  NULL,`created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,`updated_at` TIMESTAMP NOT NULL ON UPDATE CURRENT_TIMESTAMP , PRIMARY KEY (`id`));
